def ww(el):
    arr  = 1
    a = [0]*len(el)
    for i in range(len(el)-1):
        j = a[i]
        while j > 0 and el[j] != el[i+1]:
            j = a[j-1]
        if el[j] == el[i+1]:
            a[i+1] = j+1
        else:
            a[i+1] = 0
        if a[i+1] < 2:
            a = i+1
    return arr


def main():
    el=input()
    print(ww(el))
    
main()
