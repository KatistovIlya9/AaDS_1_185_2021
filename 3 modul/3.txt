def statistic(k,type):
    a = [0] * type
    for i in range(type - 1):
        j=a[i]
        while j>0 and k[i+1] != k[j]:
            j = a[j-1]
        if (k[i+1] == k[j]):
            a[i+1] = j + 1
        else:
            a[i+1] = 0
    return a
def main():
    k = input()
    type =len(k)
    a = statistic(k,type)
    if (type%(type-a[-1])==0):
        print(type//(type-a[-1]))
    else:
        print("1")


main()
